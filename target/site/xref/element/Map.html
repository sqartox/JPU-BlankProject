118 	 * Sets the map objects.
119 	 * 
120 	 * @throws Exception exception
121 	 */
122 	// Set the different Map objects
123 	public void setMapObjects() throws Exception {
124 		// Get the level Map design
125 		String map = this.getMapDesign();
126 		System.out.println(map);
127 		System.out.println(getHeight());
128 		System.out.println(getWidth());
129 		if (getHeight() >= 1 && getWidth() >= 1) {
130 			// Crossed Map in order to create all the Map Elements
131 			this.mapObjects = new Element[this.getWidth()][this.getHeight()];
132 			for (int y = 0; y < getHeight(); y++) {
133 				String[] finalMap = map.split("\n");
134 				for (int x = 0; x < getWidth(); x++) {
135 					switch (finalMap[y].toCharArray()[x]) {
136 					case 'w':
137 						mapObjects[x][y] = new Wall(x, y);
138 						break;
139 					case 'g':
140 						mapObjects[x][y] = new Dirt(x, y);
141 						break;
142 					case 'p':
143 						mapObjects[x][y] = new Player(1, x, y);
144 						break;
145 					case 's':
146 						mapObjects[x][y] = new Stone(x, y);
147 						break;
148 					case 'r':
149 						mapObjects[x][y] = new Opponent(x, y);
150 						break;
151 					case 'n':
152 						mapObjects[x][y] = new Ground(x, y);
153 						break;
154 					case 'd':
155 						mapObjects[x][y] = new Diamond(x, y);
156 						break;
157 					case 'e':
158 						mapObjects[x][y] = new Exit(x, y);
159 						break;
160 					case 'i':
161 						mapObjects[x][y] = new IndoorWall(x, y);
162 						break;
163 					default:
164 						break;
165 					}
166 					mapObjects[x][y].setMap(this);
167 				}
168 			}
169 		}
170 	}

217 	// Get all the Opponents in the Map
218 	public ArrayList<Opponent> getOpponent() {
219 		this.opponent.clear();
220 		for (int y = 0; y < getHeight(); y++) {
221 			for (int x = 0; x < getWidth(); x++) {
222 				Element element = this.getMapObjects(x, y);
223 				if (element instanceof Opponent) {
224 					if (element.getMap().getThisOpponent(x, y).isAlive()) {
225 						// Add all the Opponents to the Opponent list
226 						this.opponent.add((Opponent) this.getMapObjects(x, y));
227 					}
228 				}
229 			}
230 		}
231 		// Return all the Opponents Map objects
232 		return opponent;
233 	}
